Always wrap tool calls in a timeout with a fallback so one slow upstream does not stall the entire agent loop.

```python
import asyncio
from dataclasses import dataclass

@dataclass
class ToolResult:
    ok: bool
    data: dict | None
    error: str | None

async def call_tool_with_timeout(tool_fn, args: dict, timeout_s: float = 8.0) -> ToolResult:
    try:
        data = await asyncio.wait_for(tool_fn(args), timeout=timeout_s)
        return ToolResult(ok=True, data=data, error=None)
    except asyncio.TimeoutError:
        return ToolResult(ok=False, data=None, error="tool timeout")
    except Exception as e:
        return ToolResult(ok=False, data=None, error=str(e))

async def run_agent_step(user_query: str, tools: dict[str, callable]) -> dict:
    results = {}
    for name, fn in tools.items():
        result = await call_tool_with_timeout(fn, {"query": user_query})
        results[name] = result
        if not result.ok:
            results[name] = {"fallback": f"{name} unavailable, continuing with remaining tools"}
    return {"tools": results}
```

- Set timeouts per tool based on p99 latency, not a single global value.
- Return a structured result so the caller can distinguish timeout from other failures.
- Continue the agent loop with partial results instead of raising; the LLM can reason with what it has.
